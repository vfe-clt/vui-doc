# 安装

npm i @vfe-clt/v-ui -S

# 整体引入

import Vue from 'vue'
import VUI from '@vfe-clt/v-ui'
import '@vfe-clt/v-ui/lib/index.css'

Vue.use(VUI)

# 按需引入

安装 babel-plugin-component

npm i babel-plugin-component -D

配置.babelrc

{
  //......
  //其他配置
  //......
  "plugins": [
    [
      "component",
      {
        "libraryName": "@vfe-clt/v-ui",
        "libDir": "lib",
        "style": "index.css"
      }
    ]
  ]
}

使用时直接引入组件，不再需要显示的引入css

import Vue from 'vue'
import { Button } from '@vfe-clt/v-ui'

Vue.use(Button)
